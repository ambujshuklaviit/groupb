  if(is.na(air[i,"Ozone"])){
    air[i,"Ozone"]<- mean(air[which(air[,"Month"]==air[i,"Month"]),"Ozone"],na.rm = TRUE)
  }

    if(is.na(air[i,"Solar.R"])){

  return((x-min(x))/(max(x)-min(x)))
}
